function generarYDescargarPDF() { const { jsPDF } = window.jspdf; const doc = new jsPDF('p', 'mm', 'a4'); // ==================== CONFIGURACIÓN GENERAL ==================== const pageWidth = 210; const pageHeight = 297; const marginX = 15; const contentWidth = pageWidth - (marginX * 2); // Colores const rojo = [204, 0, 0]; const negro = [0, 0, 0]; const azul = [0, 0, 238]; const celeste = [220, 235, 245]; // Para el TOTAL final // ==================== BORDE GENERAL DE LA PÁGINA ==================== doc.setDrawColor(0, 0, 0); doc.setLineWidth(0.6); doc.rect(10, 10, 190, 277); // ==================== HEADER: LOGO ==================== // 3 barras inclinadas paralelas, muy juntas (estilo ///) const logoX = 20; const logoY = 18; const barWidth = 2.2; const barHeight = 9; const barSpacing = 1.2; // Espacio entre barras (juntas) const skew = 3; // Inclinación de las barras doc.setFillColor(0, 0, 0); for (let i = 0; i < 3; i++) { const x = logoX + (i * (barWidth + barSpacing)); // Dibujar paralelogramo inclinado usando triangles doc.triangle( x, logoY + barHeight, x + skew, logoY, x + barWidth + skew, logoY, 'F' ); doc.triangle( x, logoY + barHeight, x + barWidth + skew, logoY, x + barWidth, logoY + barHeight, 'F' ); } // Texto del logo al lado de las barras const textoLogoX = logoX + (3 * (barWidth + barSpacing)) + skew + 2; doc.setFont('helvetica', 'bold'); doc.setFontSize(18); doc.setTextColor(0, 0, 0); doc.text('modelo', textoLogoX, logoY + 6); doc.setFont('helvetica', 'normal'); doc.setFontSize(7); doc.text('equipamientos de oficinas', textoLogoX, logoY + 10); // ==================== HEADER: FECHA ==================== const hoy = new Date(); const meses = ['Enero', 'Febrero', 'Marzo', 'Abril', 'Mayo', 'Junio', 'Julio', 'Agosto', 'Setiembre', 'Octubre', 'Noviembre', 'Diciembre']; const fechaTexto = `Montevideo , ${hoy.getDate()} de ${meses[hoy.getMonth()]} del ${hoy.getFullYear()},-`; doc.setFont('helvetica', 'normal'); doc.setFontSize(11); doc.setTextColor(0, 0, 0); doc.text(fechaTexto, pageWidth - marginX, 26, { align: 'right' }); // Línea separadora gruesa debajo del header doc.setDrawColor(0, 0, 0); doc.setLineWidth(0.8); doc.line(marginX, 32, pageWidth - marginX, 32); // ==================== DATOS DEL CLIENTE ==================== let yPos = 40; // "Sres." en rojo bold doc.setFont('helvetica', 'bold'); doc.setFontSize(11); doc.setTextColor(rojo[0], rojo[1], rojo[2]); doc.text('Sres.', marginX, yPos); // REF a la derecha (mismo nivel que Sres.) doc.setFont('helvetica', 'bold'); doc.setFontSize(10); doc.setTextColor(0, 0, 0); doc.text('REF : MOBILIARIO DE OFICINA', pageWidth - marginX, yPos, { align: 'right' }); // Nombre del cliente en rojo bold yPos += 6; const nombreCliente = (document.getElementById('cliente').value || 'Cliente').toUpperCase(); doc.setFont('helvetica', 'bold'); doc.setFontSize(11); doc.setTextColor(rojo[0], rojo[1], rojo[2]); doc.text(nombreCliente, marginX, yPos); // "Presente" con subrayado exacto al ancho yPos += 6; doc.setFont('helvetica', 'bold'); doc.setFontSize(11); doc.setTextColor(0, 0, 0); const textoPresente = 'Presente'; doc.text(textoPresente, marginX, yPos); const anchoPresente = doc.getTextWidth(textoPresente); doc.setLineWidth(0.4); doc.line(marginX, yPos + 0.8, marginX + anchoPresente, yPos + 0.8); // ==================== INTRO ==================== yPos += 8; doc.setFont('helvetica', 'bold'); doc.setFontSize(10); doc.setTextColor(0, 0, 0); doc.text('De nuestra consideracion :', marginX, yPos); yPos += 6; doc.setFont('helvetica', 'normal'); doc.setFontSize(10); doc.text(' Por intermedio de la presente tenemos el', marginX, yPos); yPos += 5; doc.text('agrado de cotizar el siguiente mobiliario de oficinas :', marginX, yPos); // ==================== TABLA DE PRODUCTOS ==================== yPos += 8; // Obtener productos (asume que existen en localStorage o en el DOM) // Adaptar según cómo los tengas en tu app actual const productos = obtenerProductosDelFormulario(); // Construir filas de la tabla const filasTabla = []; let subtotal = 0; productos.forEach(p => { const totalLinea = p.cantidad * p.precio; subtotal += totalLinea; // Línea principal: "N - Descripción principal" // Líneas secundarias con indent let descripcionFormateada = `${p.cantidad} - ${p.descripcionPrincipal}`; if (p.detalles && p.detalles.length > 0) { p.detalles.forEach(d => { descripcionFormateada += `\n ${d}`; }); } if (p.material) { descripcionFormateada += `\n ${p.material}`; } filasTabla.push([ { content: descripcionFormateada, styles: { halign: 'left' } }, { content: formatearMoneda(p.precio), styles: { halign: 'right' } }, { content: formatearMoneda(totalLinea), styles: { halign: 'right' } } ]); }); // Calcular IVA y total const iva = subtotal * 0.22; const total = subtotal + iva; // Filas de totales DENTRO de la tabla filasTabla.push([ { content: '', styles: { fillColor: [255, 255, 255], lineWidth: { top: 0.5, bottom: 0, left: 0.4, right: 0.4 } } }, { content: 'sub Total U$S', styles: { halign: 'right', fontStyle: 'italic', fontSize: 9 } }, { content: formatearMoneda(subtotal), styles: { halign: 'right', fontStyle: 'bold' } } ]); filasTabla.push([ { content: '', styles: { fillColor: [255, 255, 255] } }, { content: 'IVA 22% U$S', styles: { halign: 'right', fontStyle: 'italic', fontSize: 9 } }, { content: formatearMoneda(iva), styles: { halign: 'right', fontStyle: 'bold' } } ]); filasTabla.push([ { content: '', styles: { fillColor: [255, 255, 255] } }, { content: 'TOTAL U$S', styles: { halign: 'right', fontStyle: 'bold', fontSize: 10, fillColor: celeste } }, { content: formatearMoneda(total), styles: { halign: 'right', fontStyle: 'bold', fontSize: 10, fillColor: celeste } } ]); // Renderizar tabla con autoTable doc.autoTable({ startY: yPos, head: [[ { content: 'D E S C R I P C I O N', styles: { halign: 'center' } }, { content: 'UNITARIO', styles: { halign: 'center' } }, { content: 'TOTAL', styles: { halign: 'center' } } ]], body: filasTabla, theme: 'grid', styles: { font: 'helvetica', fontSize: 9.5, textColor: [0, 0, 0], lineColor: [0, 0, 0], lineWidth: 0.4, cellPadding: 2.5, valign: 'middle' }, headStyles: { fillColor: [255, 255, 255], textColor: [0, 0, 0], fontStyle: 'bold', fontSize: 10, lineColor: [0, 0, 0], lineWidth: 0.5 }, columnStyles: { 0: { cellWidth: 115 }, 1: { cellWidth: 32, halign: 'right' }, 2: { cellWidth: 33, halign: 'right' } }, margin: { left: marginX, right: marginX } }); // ==================== CONDICIONES COMERCIALES ==================== let yCondiciones = doc.lastAutoTable.finalY + 6; const material = document.getElementById('material')?.value || 'Material melaminico'; const formaPago = document.getElementById('formaPago')?.value || '30 días fecha de factura'; const tiempoEntrega = document.getElementById('tiempoEntrega')?.value || '4 dias habiles'; const lugarEntrega = document.getElementById('lugarEntrega')?.value || 'Montevideo'; doc.setFontSize(10); doc.setFont('helvetica', 'bold'); doc.setTextColor(0, 0, 0); // MATERIAL doc.text('MATERIAL', marginX, yCondiciones); doc.setFont('helvetica', 'normal'); doc.text(`: ${material}`, marginX + 40, yCondiciones); // FORMA DE PAGO (en rojo) yCondiciones += 5.5; doc.setFont('helvetica', 'bold'); doc.text('FORMA DE PAGO', marginX, yCondiciones); doc.setFont('helvetica', 'normal'); doc.setTextColor(rojo[0], rojo[1], rojo[2]); doc.text(`: ${formaPago}`, marginX + 40, yCondiciones); // TIEMPO DE ENTREGA (en rojo) yCondiciones += 5.5; doc.setFont('helvetica', 'bold'); doc.setTextColor(0, 0, 0); doc.text('TIEMPO DE ENTREGA', marginX, yCondiciones); doc.setFont('helvetica', 'normal'); doc.setTextColor(rojo[0], rojo[1], rojo[2]); doc.text(`: ${tiempoEntrega}`, marginX + 40, yCondiciones); // LUGAR DE ENTREGA yCondiciones += 5.5; doc.setFont('helvetica', 'bold'); doc.setTextColor(0, 0, 0); doc.text('LUGAR DE ENTREGA', marginX, yCondiciones); doc.setFont('helvetica', 'normal'); doc.text(`: ${lugarEntrega},-`, marginX + 40, yCondiciones); // ==================== FIRMA ==================== yCondiciones += 10; doc.setFont('helvetica', 'bold'); doc.setFontSize(10); doc.setTextColor(0, 0, 0); doc.text('Atte. Favio Couñago', marginX, yCondiciones); // ==================== FOOTER ==================== // Línea separadora antes del footer const yFooter = pageHeight - 25; doc.setDrawColor(0, 0, 0); doc.setLineWidth(0.4); doc.line(marginX, yFooter - 4, pageWidth - marginX, yFooter - 4); // Dirección y teléfono en rojo doc.setFont('helvetica', 'bold'); doc.setFontSize(10); doc.setTextColor(rojo[0], rojo[1], rojo[2]); doc.text('Emilio Frugoni 1250 esq. Constituyente Tel 24 - 11 - 47 - 47', pageWidth / 2, yFooter, { align: 'center' }); // Website en azul subrayado doc.setFont('helvetica', 'bold'); doc.setFontSize(10); doc.setTextColor(azul[0], azul[1], azul[2]); const website = 'www.modeloequipamientos.com'; const anchoWeb = doc.getTextWidth(website); const xWeb = (pageWidth - anchoWeb) / 2; doc.text(website, xWeb, yFooter + 6); doc.setDrawColor(azul[0], azul[1], azul[2]); doc.setLineWidth(0.3); doc.line(xWeb, yFooter + 7, xWeb + anchoWeb, yFooter + 7); // ==================== GUARDAR PDF ==================== const nombreArchivo = `Modelo-${nombreCliente.replace(/\s+/g, '_')}-${hoy.getDate()}-${hoy.getMonth() + 1}-${hoy.getFullYear()}.pdf`; doc.save(nombreArchivo); } // ==================== FUNCIONES AUXILIARES ==================== function formatearMoneda(num) { return new Intl.NumberFormat('es-UY', { minimumFractionDigits: 2, maximumFractionDigits: 2 }).format(num); } function obtenerProductosDelFormulario() { // Adaptar esta función a tu estructura actual. // Debe retornar un array de objetos con esta estructura: // { // cantidad: 1, // descripcionPrincipal: "Mesa de Reuniones rectangular de 275 x 115 x 75 de alto", // detalles: ["toda de 36mm de espesor ."], // material: "Material melaminico CEDRO NATURE", // precio: 375.00 // } const productos = []; const filas = document.querySelectorAll('.producto-fila'); filas.forEach(fila => { const cantidad = parseInt(fila.querySelector('.cantidad')?.value) || 1; const descripcion = fila.querySelector('.descripcion')?.value || ''; const precio = parseFloat(fila.querySelector('.precio')?.value) || 0; // Separar descripción en líneas (usa \n o | como separador) const lineas = descripcion.split(/\n|\|/).map(l => l.trim()).filter(l => l); productos.push({ cantidad, descripcionPrincipal: lineas[0] || descripcion, detalles: lineas.slice(1, -1), material: lineas.length > 1 ? lineas[lineas.length - 1] : '', precio }); }); return productos; }
